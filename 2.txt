git init -- служит для инициализации пустого репозитория

git status [-s||--short]
    проверка статуса  репозитории (файлов) 
    также есть сокращенный вывод через флаг -s,
    Новые, неотслеживаемые файлы помечены ?? слева от них, файлы добавленные в отслеживаемые помечены A, отредактированные файлы помечены M и так далее. 

git diff [--staged]
    для полного просмотра изменений т.е. все изменения в файле, 
    с флагом --staged можно просмотреть что находится в коммите

git add <fileName> -- добавление в индекс
git add . 
    добавляет все файлы из рабочей директории в index

git show <commitHash>
    просмотреть текущий коммит, можно также просмотреть определенный коммит указ его хеш в commitHash

git show --pretty=fuller
    флаг используется для 

git reset HEAD [catalogs|files]
    команда для отмены добавления файла из индекса
stx: 
    git status
    git reset HEAD index.html

git commit [<paths>|-a|-am|-v|--amend] 
    -- Важно!! Команды ниже работают если файлы про индексированы
    > git commit -a
        stx: git commit -am "message"
    > git commit <paths>
        коммит для определенного файла по указанному пути
    > git commit -v
        для просмотра изменений в репозитории
    -- Для обхода ограничений можно создать алиас
    > git config --global alias.commitall "!git add .;git commit"
        stx: git commitall -m "Commit message"
    > git commit --amend
        для объединения коммитов если выполнить эту команду то она перекроет впереди стоящий коммит

git rm <paths>
    команда для удаления из рабочей папки и репозитории
    > -r 
        флаг для удаления папки
        stx: git rm -r <directory>
    > -f
        флаг для удаления с пропуском предупреждений
        stx: git rm -f <paths>
    > --cashed
        флаг для удаления из индекса
        stx: git rm --cashed <paths>

git mv <old> <new>
    команда для переименования
